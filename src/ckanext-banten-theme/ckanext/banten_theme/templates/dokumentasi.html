    {# --- 2. Akses via API ------------------------------------- #}

    2. {{ _('Akses via API') }}

    {{ _('Satu Data Banten menyediakan REST API berbasis CKAN Action API. Seluruh dataset publik dapat diakses tanpa autentikasi. API Token hanya diperlukan untuk operasi tulis (membuat/mengubah dataset) atau membaca data privat.') }}

    {{ _('Endpoint utama') }}

    BASE URL GET
    https://satudata.bantendev.id/api/3/action/{action}
    Action{{ _('Deskripsi') }}
    package_search{{ _('Pencarian dataset (mendukung query, filter, paging).') }}
    package_show{{ _('Detail satu dataset berdasarkan id atau name.') }}
    package_list{{ _('Daftar nama dataset publik.') }}
    organization_list{{ _('Daftar OPD/instansi penyedia data.') }}
    group_list{{ _('Daftar kategori (group) data.') }}
    tag_list{{ _('Daftar seluruh tag/kata kunci.') }}
    resource_show{{ _('Metadata satu resource (file/CSV/dll) di dalam dataset.') }}

    {{ _('Contoh: pencarian dataset') }}

    curl 'https://satudata.bantendev.id/api/3/action/package_search?q=populasi&rows=1'
    {
  "success": true,
  "result": {
    "count": 1810,
    "results": [{
      "title": "Dokumen Hasil Fasilitasi ...",
      "name":  "dokumen-hasil-fasilitasi-...",
      "organization": { "title": "Biro Organisasi & RB" },
      "metadata_modified": "2026-04-17T03:41:28",
      "resources": [ ... ]
    }]
  }
}

    {{ _('Contoh: detail dataset') }}

    curl 'https://satudata.bantendev.id/api/3/action/package_show?id=<dataset_name_atau_id>'

    {{ _('Autentikasi (opsional)') }}

    {{ _('Kirim API Token via header HTTP berikut hanya jika Anda perlu mengakses data privat atau melakukan write action:') }}

    curl 'https://satudata.bantendev.id/api/3/action/package_show?id=...' \
  -H 'Authorization: <CKAN_API_TOKEN>'
    {{ _('Tip:') }} {{ _('Login → klik nama Anda di kanan atas → tab "API Tokens" untuk membuat token baru. Token hanya ditampilkan satu kali, simpan dengan aman.') }}
    {{ _('Catatan:') }} {{ _('Dokumentasi lengkap CKAN Action API tersedia di') }} docs.ckan.org/en/2.11/api.
